功能介绍:

1.支持是否拖拽以及支持触摸设备
2.支持滚轮滑动
3.支持自定义速度
4.支持jquery.easing库效果默认swing
5.支持是否自动滑动，和滑动间隔
6.支持纵向和横向滑动
7.支持绑定上一页和下一页绑定按钮
8.支持是否显示页码
9.支持每页显示几个单元

调用方法:

$("#test").slider({
	size:1,//显示区域显示多少片
	prev:"#prev",//绑定上一组ID
	next:"#next",//绑定下一组ID
	prev_disable_class:"prev-disabled",//当上一组没有时的样式
	next_disable_class:"next-disabled",//当下一组没有时的样式
	pager:"#pager",//绑定页码，当不设置不显示页码
	pager_event:'click',//页码相应方式，默认是mouseover
	easing:'easeOutCirc',//滑动效果算法插件，需要引入jquery.easing.js
	mousewheel:true,//支持鼠标滚轮滚动
	speed:600,//速度
	prev_callback:function(page){//点上一组后回调
		var me=$("#qa-say").find("li").eq(page).find(".accordion-active");
		show_answer(me.find("em").text(),me.data("a"),me.data("b"),me.data("c"));
	},
	next_callback:function(page){//点下一组后回调
		var me=$("#qa-say").find("li").eq(page).find(".accordion-active");
		show_answer(me.find("em").text(),me.data("a"),me.data("b"),me.data("c"));
	},
	pager_callback:function(page){//点页码回调
		var me=$("#qa-say").find("li").eq(page).find(".accordion-active");
		show_answer(me.find("em").text(),me.data("a"),me.data("b"),me.data("c"));
	}
});
